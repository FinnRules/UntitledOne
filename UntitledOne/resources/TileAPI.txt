#tiles with no effects:
class <Name>(MapTile):
  def intro_text(self):
    return """Room intro text""" #triple quote supports multi line strings

  def modify_player(self, the_player):
    pass

#tiles with items, requires grab command
class <Name>(GrabLootRoom):
  def __init__(self, x, y):
    super().__init__(x, y, <item>) #<item> should be a plain object like items.Scalpel(), include param and format as a var
  
  def intro_text(self):
    return """Room text"""

#room that instantly gives an item (discouraged)
class <Name>(LootRoom):
   def __init__(self, x, y):
    super().__init__(x, y, <item>)

  def intro_text(self):
    return """Room text"""

#mob room
class <Name>(MobRoom):
  def __init__(self, x, y):
    super().__init__(x, y, <enemy>) #same formatting as item, eg enemies.Guard()
    
  def intro_text(self):
    if self.enemy.is_alive():
      return """text if enemy is alive"""
    else:
      return """text if enemy is dead"""

#end room (should be used sparingly)
class <Name>(MapTile):
  def intro_text(self):
    return """Room text"""
  
  def modify_player(self, player):
    player.victory = True

#instakill room
class <Name>(MapTile):
  def intro_text(self):
    return """Room text"""

  def modify_player(self, player):
    player.hp = 0 #alternativily if you want to just dock some amount of damage: player.hp -= 10 etc
